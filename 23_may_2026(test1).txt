-- 1. [JOINS] List the name of each employee along with their job title and department name.
SELECT e.name, j.job_title, d.dept_name FROM Employees e JOIN Jobs j ON e.job_id = j.job_id JOIN Departments d ON e.dept_id = d.dept_id;

-- 2. [SUBQUERY] Find all employees whose salary is higher than the average salary of the IT department.
SELECT name, salary FROM Employees
WHERE salary > (
     SELECT AVG(salary) FROM Employee WHERE dept_id = ( SELECT dept_id FROM Departments WHERE dept_name = 'IT')
);

-- 3. [GROUP BY] Calculate the total salary expenditure for each department name.
SELECT d.dept_name, SUM(e.salary) AS total_salary
FROM Departments d
JOIN Employees e ON d.dept_id = e.dept_id
GROUP BY d.dept_name;

-- 4. [HAVING] Display departments that have more than 2 employees and an average salary above 70000.
SELECT d.dept_name, COUNT(e.emp_id) AS employee_count,
       AVG(e.salary) AS avg_salary
FROM Departments d
JOIN Employees e ON d.dept_id = e.dept_id
GROUP BY d.dept_name
HAVING COUNT(e.emp_id) > 2
   AND AVG(e.salary) > 70000;

-- 5. [SELF JOIN] Retrieve employee names and their manager names.
SELECT e.name AS employee,
       m.name AS manager
FROM Employees e
LEFT JOIN Employees m
ON e.manager_id = m.emp_id;

-- 6. [ORDER BY] Find the top 3 highest-paid employees.
SELECT name, salary FROM Employees ORDER BY salary DESC LIMIT 3;

-- 7. [JOINS + WHERE] Find all Developers working in San Francisco.
SELECT e.name, j.job_title, d.location
FROM Employees e
JOIN Jobs j ON e.job_id = j.job_id
JOIN Departments d ON e.dept_id = d.dept_id
WHERE j.job_title = 'Developer'
  AND d.location = 'San Francisco';


-- 8. [GROUP BY] Find the number of employees hired each year.
SELECT YEAR(hire_date) , COUNT(*) FROM Employees GROUP BY YEAR(hire_date);


-- 9. [HAVING] List job titles where the maximum employee salary is less than 100000.
SELECT j.job_title, MAX(e.salary) AS max_salary FROM Jobs j
JOIN Employees e ON j.job_id = e.job_id GROUP BY j.job_title HAVING MAX(e.salary) < 100000;



-- 14. [JOINS] Find the job title and salary of employee with ID 504.
SELECT e.name, j.job_title, e.salary FROM Employees e
JOIN Jobs j ON e.job_id = j.job_id  WHERE e.emp_id = 504;

-- 15. [GROUP BY] Minimum salary under each manager.
SELECT manager_id,
       MIN(salary) AS minimum_salary
FROM Employees
WHERE manager_id IS NOT NULL
GROUP BY manager_id;

-- 18. [JOIN + ORDER BY] Employee names and hire dates sorted by department and salary.
SELECT e.name, e.hire_date, d.dept_name, e.salary
FROM Employees e JOIN Departments d ON e.dept_id = d.dept_id ORDER BY d.dept_name ASC, e.salary DESC;

-- 19. [HAVING] Managers who manage more than 2 employees.
SELECT manager_id,  COUNT(*) AS employees_managed FROM Employees WHERE manager_id IS NOT NULL GROUP BY manager_id HAVING COUNT(*) > 2;

-- 20. [SUBQUERY] Employee(s) with the third-highest salary.
SELECT name, salary
FROM Employees
WHERE salary = ( SELECT DISTINCT salary FROM Employees 
ORDER BY salary DESC LIMIT 1 OFFSET 2
);

-- 21. [LEFT JOIN] List all job titles and number of employees in each job.
SELECT j.job_title, COUNT(e.emp_id) AS employee_count
FROM Jobs j LEFT JOIN Employees e ON j.job_id = e.job_id GROUP BY j.job_title;

-- 22. [JOIN + LIKE] Finance employees whose names contain 'i'.
SELECT e.name FROM Employees e JOIN Departments d ON e.dept_id = d.dept_id
WHERE d.dept_name = 'Finance' AND e.name LIKE '%i%';

-- 24. [SUBQUERY] Department paying the highest total salary.
SELECT dept_name FROM DepartmentsWHERE dept_id = ( SELECT dept_id FROM Employees
    GROUP BY dept_id ORDER BY SUM(salary) DESC LIMIT 1
);


-- 25. [JOINS] Employees who report to Charlie.
SELECT e.name AS employee_name FROM Employees e JOIN Employees m ON e.manager_id = m.emp_id
WHERE m.name = 'Charlie';
