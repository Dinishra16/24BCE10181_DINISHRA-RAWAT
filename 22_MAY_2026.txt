-- Cross Joint :-  A CROSS JOIN in MySQL is used to combine every row of the first table with every row of the second table.
-- This creates a Cartesian Product.

                                                            														
                                                            
use vitcampus;
create table department(
did int primary key,
dname varchar(25) not null unique 
);
insert into department value (101,'CSE'),(102,'ECE'),(103,'CIVIL'),(106,'AERO'),(107,'MECH');

select*from department order by did asc;

create table HOD
(Hid int primary key,
hname varchar(20) not null unique);

insert into HOD values
(101,'Arun'),(102,'Jayanth'),(104,'Karthik'),
(105,'Murali');

select * from HOD;

SELECT * FROM department CROSS JOIN hod order by did asc;


-- Inner join :-- is used to get matching rows from two tables based on a related column. 

SELECT * FROM department inner join hod where department.did = hod.hid ;


-- Natural join :- it checks column anme and data types 

select*from department natural join hod;


-- outer join :- 
--  ~ ~ left outer join
select*from department left outer join hod on ( department.did = hod.hid ) order by did asc ;

-- ~ ~ right outer join

select*from department right outer join hod on ( department.did = hod.hid ) order by did asc ;

-- ~ ~ full outer join

select*from department full join hod on ( department.did = hod.hid ) order by did asc ;
